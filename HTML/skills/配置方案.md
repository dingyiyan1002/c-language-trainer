## 🎯 用途

这个目录存放所有可视化 HTML 的**模板和组件**，用于：

1. **统一规范** - 所有 HTML 使用相同的模板
2. **快速生成** - 基于模板快速生成新题目
3. **质量保证** - 确保每题都符合 v5.3 规范
4. **复用组件** - 代码编辑器、执行光球等组件可复用

### 4. skills/components/CodeEditor.js（组件示例）

```javascript
// CodeEditor 组件 - v5.3 规范
// 四层结构：行号、语法高亮、执行光球、输入层

function CodeEditor({ code, currentLine, executedLines = [] }) {
  const [scrollTop, setScrollTop] = useState(0);
  
  return (
    <div className="code-editor" style={{ position: 'relative' }}>
      {/* Layer 1: 行号 */}
      <div className="line-numbers" style={{
        position: 'absolute',
        left: 0,
        width: '50px',
        background: '#1e293b',
        color: '#64748b',
        textAlign: 'right',
        padding: '10px 5px'
      }}>
        {code.split('\n').map((_, i) => (
          <div key={i} className={`line-number ${i + 1 === currentLine ? 'active' : ''}`}>
            {i + 1}
          </div>
        ))}
      </div>
      
      {/* Layer 2: 语法高亮背景 */}
      <pre className="highlight-layer" style={{
        position: 'absolute',
        inset: 0,
        paddingLeft: '60px',
        color: '#e2e8f0',
        fontFamily: 'JetBrains Mono, monospace',
        fontSize: '14px',
        lineHeight: '1.6'
      }}>
        <code dangerouslySetInnerHTML={{ __html: highlightCode(code) }} />
      </pre>
      
      {/* Layer 3: 执行光球 */}
      {executedLines.includes(currentLine) && (
        <div className="exec-orb" style={{
          position: 'absolute',
          left: '55px',
          top: `${(currentLine - 1) * 22.4}px`,
          width: '20px',
          height: '20px',
          borderRadius: '50%',
          background: 'linear-gradient(135deg, #22c55e, transparent)',
          animation: 'orbGlow 1s infinite'
        }} />
      )}
      
      {/* Layer 4: 透明输入层 */}
      <textarea
        value={code}
        readOnly
        style={{
          position: 'absolute',
          inset: 0,
          paddingLeft: '60px',
          background: 'transparent',
          color: 'transparent',
          caretColor: '#22d3ee',
          fontFamily: 'JetBrains Mono, monospace',
          fontSize: '14px',
          lineHeight: '1.6',
          outline: 'none'
        }}
      />
    </div>
  );
}

// 语法高亮函数
function highlightCode(code) {
  return code
    .replace(/(int|char|float|double|void|return|if|else|for|while)/g, '<span style="color: #c084fc">$1</span>')
    .replace(/(printf|scanf|malloc|free)/g, '<span style="color: #22d3ee">$1</span>')
    .replace(/("[^"]*")/g, '<span style="color: #4ade80">$1</span>')
    .replace(/(\d+)/g, '<span style="color: #fb923c">$1</span>')
    .replace(/(\/\/.*)/g, '<span style="color: #64748b">$1</span>');
}

export default CodeEditor;
```

### 2. scripts/generate-html.js（生成脚本）

```javascript
const fs = require('fs');
const path = require('path');

// 读取配置
const config = require('../HTML/skills/config.json');

// 读取模板
const template = fs.readFileSync(
  path.join(__dirname, '../HTML/skills/templates/basic-template.html'),
  'utf8'
);

// 生成 HTML
function generateHTML(lesson, chapter) {
  let html = template;
  
  // 替换变量
  html = html.replace('{{lessonId}}', lesson.id);
  html = html.replace('{{title}}', lesson.title);
  html = html.replace('{{content}}', lesson.content);
  html = html.replace('{{code}}', extractCode(lesson.content));
  html = html.replace('{{variables}}', JSON.stringify(lesson.variables));
  html = html.replace('{{steps}}', JSON.stringify(lesson.steps));
  
  // 保存文件
  const outputPath = path.join(
    __dirname,
    `../HTML/${config.output.chapters[chapter]}/${chapter}_${lesson.id}.html`
  );
  
  fs.writeFileSync(outputPath, html, 'utf8');
  console.log(`Generated: ${outputPath}`);
}

// 批量生成
function generateAll() {
  const chapters = require('../src/data/chapters');
  
  for (const [chapterId, chapter] of Object.entries(chapters)) {
    chapter.lessons.forEach(lesson => {
      generateHTML(lesson, chapterId);
    });
  }
}

generateAll();
```
