```c
ListNode* reverseList(ListNode* head) {
    ListNode *prev = NULL, *curr = head;
    while (curr) {
        ListNode* next = curr->next;
        curr->next = prev;
        prev = curr;
        curr = next;
    }
    return prev;
}

// 2. 快速排序
void quickSort(int arr[], int l, int r) {
    if (l >= r) return;
    int i = l, j = r, pivot = arr[l];
    while (i < j) {
        while (i < j && arr[j] >= pivot) j--;
        arr[i] = arr[j];
        while (i < j && arr[i] <= pivot) i++;
        arr[j] = arr[i];
    }
    arr[i] = pivot;
    quickSort(arr, l, i-1);
    quickSort(arr, i+1, r);
}
```

#### 经典面试题
```
Q1: 进程和线程的区别？
A: 
- 进程是资源分配最小单位，线程是 CPU 调度最小单位
- 进程间内存独立，线程共享进程内存
- 进程切换开销大，线程切换开销小
- 进程通信需要 IPC，线程可直接读写数据

Q2: 什么是死锁？如何预防？
A: 
死锁四个必要条件：
1. 互斥条件
2. 请求与保持
3. 不剥夺条件
4. 循环等待

预防策略：破坏任一条件
- 一次性分配所有资源（破坏请求与保持）
- 允许抢占（破坏不剥夺）
- 资源有序分配（破坏循环等待）
```

```c
// 1. 二分查找（必须熟练）
int binarySearch(int arr[], int n, int target) {
    int l = 0, r = n - 1;
    while (l <= r) {
        int mid = l + (r - l) / 2;
        if (arr[mid] == target) return mid;
        else if (arr[mid] < target) l = mid + 1;
        else r = mid - 1;
    }
    return -1;
}

// 2. 判断 2 的幂（位运算经典）
bool isPowerOfTwo(int n) {
    return n > 0 && (n & (n - 1)) == 0;
}

// 3. 统计二进制中 1 的个数
int countOnes(int n) {
    int count = 0;
    while (n) {
        n &= (n - 1);  // 消除最低位的 1
        count++;
    }
    return count;
}
```
